  symforce_slam
  ${SYMFORCE_LIBRARY_TYPE}
  ${SYMFORCE_SLAM_SOURCES}
  ${SYMFORCE_SLAM_HEADERS}
)

target_link_libraries(symforce_slam
  symforce_gen
  ${SYMFORCE_EIGEN_TARGET}
)
